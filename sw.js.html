const CACHE_NAME = 'satshadow-v2.5'; const ASSETS = [ './', './index.html', './manifest.json' ]; // تثبيت الكاش self.addEventListener('install', (e) => { e.waitUntil( caches.open(CACHE_NAME).then((cache) => { return cache.addAll(ASSETS); }).then(() => self.skipWaiting()) ); }); // تفعيل وتحديث الكاش القديم self.addEventListener('activate', (e) => { e.waitUntil( caches.keys().then((keys) => { return Promise.all( keys.map((key) => { if (key !== CACHE_NAME) { return caches.delete(key); } }) ); }).then(() => self.clients.claim()) ); }); // التعامل مع الطلبات أوفلاين (الشرط الأساسي لكروم) self.addEventListener('fetch', (e) => { e.respondWith( caches.match(e.request).then((cachedResponse) => { if (cachedResponse) { return cachedResponse; } return fetch(e.request).catch(() => { // لو مفيش نت والحاجة مش في الكاش if (e.request.mode === 'navigate') { return caches.match('./index.html'); } }); }) ); });
